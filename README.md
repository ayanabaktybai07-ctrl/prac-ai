def analyze_user_request(user_input):
    # Критикалық триггерлік сөздер тізімі (Массив)
    trigger_words = ["көмек", "кризис", "нашар", "проблема", "ақау"]
    
    print("Сұраныс сәтті қабылданды. Алгоритм іске қосылды...")
    
    is_critical = False
    
    # Тізім элементтерін цикл арқылы аралап шығу (Итерация)
    for word in trigger_words:
        # Кірістірілген шартты оператор арқылы мәтінді тексеру
        if word in user_input.lower():
            is_critical = True
            break  # Сәйкестік табылса, уақыт пен жадыны үнемдеу үшін циклді тоқтату
            
    # Соңғы шешімді қабылдау блогы
    if is_critical:
        return "Жүйелік хабарлама: КРИТИКАЛЫҚ жағдай! Шұғыл көмек көрсету модулі іске қосылды."
    else:
        return "Жүйелік хабарлама: Сұраныс қалыпты. Стандартты өңдеу режимі белсенді."

# Тестілеу бөлімі
test_text = "Маған шұғыл көмек қажет, жүйеде белгісіз ақау пайда болды."
result = analyze_user_request(test_text)
print(result)
